// ── PANE / WINDOW DRAGGING & RESIZING ───────────────────────────
// While dragging anything that can pass over the live preview iframe, mouse
// events that land on top of the iframe get delivered to the iframe's own
// document instead of bubbling up to ours — which is what made dragging feel
// "stuck"/harder whenever the cursor crossed over the preview. Disabling
// pointer-events on the iframe for the duration of any drag fixes that.
function disablePreviewPointerEvents() {
  $preview.style.pointerEvents = 'none';
}
function enablePreviewPointerEvents() {
  $preview.style.pointerEvents = '';
}

function setupHResizer() {
  $hResizer.addEventListener('mousedown', e => {
    e.preventDefault();
    const startX = e.clientX;
    const startW = $previewPane.getBoundingClientRect().width;
    document.body.style.userSelect = 'none';
    $hResizer.classList.add('dragging');
    disablePreviewPointerEvents();

    function onMove(ev) {
      const dx = startX - ev.clientX; // drag left => grows preview, drag right => shrinks preview
      let newW = startW + dx;
      const maxW = $leftCol.getBoundingClientRect().width - 200; // leave room for editor
      newW = Math.max(200, Math.min(maxW, newW));
      $previewPane.style.width = newW + 'px';
    }
    function onUp() {
      document.removeEventListener('mousemove', onMove);
      document.removeEventListener('mouseup', onUp);
      document.body.style.userSelect = '';
      $hResizer.classList.remove('dragging');
      enablePreviewPointerEvents();
      localStorage.setItem('previewWidth', $previewPane.style.width);
    }
    document.addEventListener('mousemove', onMove);
    document.addEventListener('mouseup', onUp);
  });
}

function makeDraggable(handle, panel) {
  handle.addEventListener('mousedown', e => {
    if (e.target.closest('button')) return; // don't drag when clicking a control in the header
    e.preventDefault();
    const startX = e.clientX, startY = e.clientY;
    const rect = panel.getBoundingClientRect();
    const startLeft = rect.left, startTop = rect.top;
    document.body.style.userSelect = 'none';
    disablePreviewPointerEvents();

    function onMove(ev) {
      let left = startLeft + (ev.clientX - startX);
      let top  = startTop  + (ev.clientY - startY);
      left = Math.max(0, Math.min(window.innerWidth - panel.offsetWidth, left));
      top  = Math.max(40, Math.min(window.innerHeight - 40, top));
      panel.style.left = left + 'px';
      panel.style.top  = top  + 'px';
    }
    function onUp() {
      document.removeEventListener('mousemove', onMove);
      document.removeEventListener('mouseup', onUp);
      document.body.style.userSelect = '';
      enablePreviewPointerEvents();
      saveAIPaneState();
    }
    document.addEventListener('mousemove', onMove);
    document.addEventListener('mouseup', onUp);
  });
}

function makeResizable(handle, panel) {
  handle.addEventListener('mousedown', e => {
    e.preventDefault(); e.stopPropagation();
    const startX = e.clientX, startY = e.clientY;
    const rect = panel.getBoundingClientRect();
    const startW = rect.width, startH = rect.height;
    document.body.style.userSelect = 'none';
    disablePreviewPointerEvents();

    function onMove(ev) {
      const w = Math.max(280, Math.min(window.innerWidth  - rect.left, startW + (ev.clientX - startX)));
      const h = Math.max(220, Math.min(window.innerHeight - rect.top,  startH + (ev.clientY - startY)));
      panel.style.width  = w + 'px';
      panel.style.height = h + 'px';
    }
    function onUp() {
      document.removeEventListener('mousemove', onMove);
      document.removeEventListener('mouseup', onUp);
      document.body.style.userSelect = '';
      enablePreviewPointerEvents();
      if (!aiMinimized) saveAIPaneState();
    }
    document.addEventListener('mousemove', onMove);
    document.addEventListener('mouseup', onUp);
  });
}

async function callAI(userMsg) {
  if (!API_KEY) {
    addMsg('ai', '⚠️ No API key set. Open this file and paste your Groq key into `API_KEY` at the top of the script.');
    return;
  }

  const isEdit = editModeOn;
  const codeSnapshot = $code.value; // snapshot used both for the prompt and for diffing the reply

  addMsg('user', userMsg);
  history.push({ role: 'user', content: userMsg });

  // Loading bubble
  const loadEl = document.createElement('div');
  loadEl.className = 'msg ai';
  loadEl.innerHTML = `<div class="msg-label">AI Tutor</div><div class="msg-body"><span class="spin"></span>Thinking…</div>`;
  $chat.appendChild(loadEl);
  $chat.scrollTop = $chat.scrollHeight;

  try {
    let dynamicSystem = SYSTEM;

    if (isEdit) {
      // Edit mode always sends the full, untruncated file — we need every line to diff and to apply.
      dynamicSystem += `\n\nThe user's HTML editor currently contains the following COMPLETE file:\n\`\`\`html\n${codeSnapshot}\n\`\`\`` + EDIT_SYSTEM_SUFFIX;
    } else if (codeContextOn) {
      const codeSnippet = codeSnapshot.length > 8000 ? codeSnapshot.slice(0, 8000) + '\n<!-- …truncated… -->' : codeSnapshot;
      dynamicSystem += `\n\nThe user's HTML editor currently contains the following code. It updates live as they type, so always treat this as their CURRENT code — even if they don't paste or mention it themselves:\n\`\`\`html\n${codeSnippet}\n\`\`\``;
    }

    const res = await fetch(API_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${API_KEY}` },
      body: JSON.stringify({
        model: MODELS[effort] || MODELS.high,
        max_tokens: 2048,
        temperature: 0.7,
        messages: [{ role: 'system', content: dynamicSystem }, ...history.slice(-16)]
      })
    });

    loadEl.remove();

    if (!res.ok) {
      const e = await res.json().catch(() => ({}));
      addMsg('ai', `❌ API error: ${e?.error?.message || res.status}`);
      return;
    }

    const data = await res.json();
    const reply = data?.choices?.[0]?.message?.content || '(empty response)';

    lastAI = reply;
    const histEntry = { role: 'assistant', content: reply };
    if (isEdit) { histEntry.isEdit = true; histEntry.codeSnapshot = codeSnapshot; }
    history.push(histEntry);
    if (history.length > 20) history = history.slice(-20);
    localStorage.setItem('chat', JSON.stringify(history));

    addMsg('ai', reply, isEdit ? codeSnapshot : null);

  } catch (err) {
    loadEl.remove();
    addMsg('ai', `❌ Network error: ${err.message}`);
  }
}

// ── LINE DIFF (LCS-based) ─────────────────────────────────────────
function diffLines(oldStr, newStr) {
  const oldLines = oldStr.split('\n');
  const newLines = newStr.split('\n');
  const m = oldLines.length, n = newLines.length;

  // Guard against pathological sizes — fall back to a trivial whole-file diff.
  if (m * n > 4_000_000) {
    const ops = [];
    oldLines.forEach(l => ops.push({ type: 'del', line: l }));
    newLines.forEach(l => ops.push({ type: 'add', line: l }));
    return ops;
  }

  const dp = Array.from({ length: m + 1 }, () => new Int32Array(n + 1));
  for (let i = m - 1; i >= 0; i--) {
    for (let j = n - 1; j >= 0; j--) {
      dp[i][j] = oldLines[i] === newLines[j] ? dp[i + 1][j + 1] + 1 : Math.max(dp[i + 1][j], dp[i][j + 1]);
    }
  }

  const ops = [];
  let i = 0, j = 0;
  while (i < m && j < n) {
    if (oldLines[i] === newLines[j]) { ops.push({ type: 'same', line: oldLines[i] }); i++; j++; }
    else if (dp[i + 1][j] >= dp[i][j + 1]) { ops.push({ type: 'del', line: oldLines[i] }); i++; }
    else { ops.push({ type: 'add', line: newLines[j] }); j++; }
  }
  while (i < m) { ops.push({ type: 'del', line: oldLines[i] }); i++; }
  while (j < n) { ops.push({ type: 'add', line: newLines[j] }); j++; }
  return ops;
}

function renderDiffBlock(oldCode, newCode) {
  const ops = diffLines(oldCode, newCode);
  const CONTEXT = 3;
  const n = ops.length;
  const show = new Array(n).fill(false);
  let addCount = 0, delCount = 0;
  ops.forEach((op, idx) => {
    if (op.type === 'add') addCount++;
    if (op.type === 'del') delCount++;
    if (op.type !== 'same') {
      for (let k = Math.max(0, idx - CONTEXT); k <= Math.min(n - 1, idx + CONTEXT); k++) show[k] = true;
    }
  });

  let body = '';
  let oldLn = 1, newLn = 1, i = 0;
  while (i < n) {
    if (!show[i]) {
      let j = i;
      while (j < n && !show[j]) j++;
      const count = j - i;
      body += `<div class="diff-collapsed">⋯ ${count} unchanged line${count > 1 ? 's' : ''} ⋯</div>`;
      for (let k = i; k < j; k++) { oldLn++; newLn++; }
      i = j;
      continue;
    }
    const op = ops[i];
    if (op.type === 'same') {
      body += `<div class="diff-line ctx"><span class="diff-ln">${oldLn}</span><span class="diff-code">${esc(op.line)}</span></div>`;
      oldLn++; newLn++;
    } else if (op.type === 'del') {
      body += `<div class="diff-line del"><span class="diff-ln">${oldLn}</span><span class="diff-code">− ${esc(op.line)}</span></div>`;
      oldLn++;
    } else {
      body += `<div class="diff-line add"><span class="diff-ln">${newLn}</span><span class="diff-code">+ ${esc(op.line)}</span></div>`;
      newLn++;
    }
    i++;
  }

  const id = editIdCounter++;
  pendingEdits[id] = { newCode };

  if (addCount === 0 && delCount === 0) {
    return `<div class="diff-wrap" data-edit-id="${id}"><div class="diff-header"><span>📝 Proposed changes</span><span class="diff-stats">no changes detected</span></div></div>`;
  }

  return `<div class="diff-wrap" data-edit-id="${id}">
    <div class="diff-header"><span>📝 Proposed changes</span><span class="diff-stats"><span class="add-count">+${addCount}</span> <span class="del-count">−${delCount}</span></span></div>
    <div class="diff-body">${body}</div>
    <div class="diff-actions">
      <button class="diff-apply" onclick="applyEdit(${id})">✅ Apply Changes</button>
      <button class="diff-discard" onclick="discardEdit(${id})">✖ Discard</button>
    </div>
  </div>`;
}

function applyEdit(id) {
  const data = pendingEdits[id];
  if (!data || data.done) return;
  $code.value = data.newCode;
  localStorage.setItem('code', $code.value);
  updateLines();
  runCode();
  data.done = true;
  markDiffDone(id, 'applied');
}

function discardEdit(id) {
  const data = pendingEdits[id];
  if (!data || data.done) return;
  data.done = true;
  markDiffDone(id, 'discarded');
}

function markDiffDone(id, state) {
  const wrap = document.querySelector(`.diff-wrap[data-edit-id="${id}"]`);
  if (!wrap) return;
  wrap.classList.add('diff-' + state);
  const actions = wrap.querySelector('.diff-actions');
  if (actions) {
    actions.innerHTML = state === 'applied'
      ? '<span class="diff-status applied">✅ Applied to editor</span>'
      : '<span class="diff-status discarded">✖ Discarded</span>';
  }
}

// ── CHAT RENDERING ───────────────────────────────────────────────
function addMsg(role, text, editSnapshot) {
  const div = document.createElement('div');
  div.className = `msg ${role}`;
  const label = role === 'user' ? 'You' : 'AI Tutor';
  let body;
  if (role === 'user') {
    body = `<div class="msg-body">${esc(text)}</div>`;
  } else if (editSnapshot != null) {
    body = `<div class="msg-body">${mdWithDiff(text, editSnapshot)}</div>`;
  } else {
    body = `<div class="msg-body">${md(text)}</div>`;
  }
  div.innerHTML = `<div class="msg-label">${label}</div>${body}`;
  $chat.appendChild(div);
  $chat.scrollTop = $chat.scrollHeight;

  // copy buttons in plain code blocks (not diff blocks)
  div.querySelectorAll('.code-wrap button').forEach(btn => {
    btn.onclick = () => {
      const pre = btn.closest('.code-wrap').querySelector('pre');
      navigator.clipboard.writeText(pre.textContent).then(() => {
        btn.textContent = 'Copied!';
        setTimeout(() => btn.textContent = 'Copy', 1500);
      });
    };
  });
}

// ── MARKDOWN ─────────────────────────────────────────────────────
function esc(s) {
  return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

function splitCodeParts(text) {
  const parts = [];
  const re = /```(\w*)\n?([\s\S]*?)```/g;
  let last = 0, m;
  while ((m = re.exec(text)) !== null) {
    if (m.index > last) parts.push({ t: 'text', v: text.slice(last, m.index) });
    parts.push({ t: 'code', lang: m[1] || 'html', v: m[2].trimEnd() });
    last = re.lastIndex;
  }
  parts.push({ t: 'text', v: text.slice(last) });
  return parts;
}

function mdText(v) {
  let t = esc(v);
  t = t.replace(/^### (.+)$/gm, '<h3>$1</h3>');
  t = t.replace(/^## (.+)$/gm,  '<h2>$1</h2>');
  t = t.replace(/^# (.+)$/gm,   '<h1>$1</h1>');
  t = t.replace(/\*\*(.+?)\*\*/g, '<strong>$1</strong>');
  t = t.replace(/\*(.+?)\*/g,     '<em>$1</em>');
  t = t.replace(/`([^`]+)`/g,     '<code>$1</code>');
  t = t.replace(/^&gt; (.+)$/gm,  '<blockquote>$1</blockquote>');
  t = t.replace(/^[\*\-] (.+)$/gm,'<li>$1</li>');
  t = t.replace(/^\d+\. (.+)$/gm, '<li>$1</li>');
  t = t.replace(/\n{2,}/g, '</p><p>');
  t = t.replace(/\n/g, '<br>');
  return `<p>${t}</p>`;
}

function md(text) {
  return splitCodeParts(text).map(p => {
    if (p.t === 'code') {
      return `<div class="code-wrap"><div class="code-top"><span>${esc(p.lang)}</span><button>Copy</button></div><pre>${esc(p.v)}</pre></div>`;
    }
    return mdText(p.v);
  }).join('');
}

function mdWithDiff(text, oldCode) {
  let usedDiff = false;
  return splitCodeParts(text).map(p => {
    if (p.t === 'code') {
      if (!usedDiff) {
        usedDiff = true;
        return renderDiffBlock(oldCode, p.v);
      }
      return `<div class="code-wrap"><div class="code-top"><span>${esc(p.lang)}</span><button>Copy</button></div><pre>${esc(p.v)}</pre></div>`;
    }
    return mdText(p.v);
  }).join('');
}
