  /* ═══════════════════════════════════════════════════════
     TAB STATE
  ═══════════════════════════════════════════════════════ */
  let tabs     = [];
  let activeId = null;

  const STORAGE_TABS   = 'html_editor_tabs';
  const STORAGE_ACTIVE = 'html_editor_active';

  function genId() { return 't' + Date.now() + Math.random().toString(36).slice(2, 6); }

  function saveTabs() {
    localStorage.setItem(STORAGE_TABS,   JSON.stringify(tabs));
    localStorage.setItem(STORAGE_ACTIVE, activeId);
  }

  function loadTabs() {
    try {
      const raw = localStorage.getItem(STORAGE_TABS);
      if (raw) {
        tabs = JSON.parse(raw);
        activeId = localStorage.getItem(STORAGE_ACTIVE) || tabs[0]?.id || null;
        return;
      }
    } catch(e) {}
    const legacy = localStorage.getItem('html_code') || '<h1>Hello World</h1>';
    const id = genId();
    tabs = [{ id, name: 'Tab 1', code: legacy }];
    activeId = id;
    saveTabs();
  }

  function activeTab() { return tabs.find(t => t.id === activeId) || null; }

  /* ═══════════════════════════════════════════════════════
     TAB UI + DRAG-TO-REORDER
  ═══════════════════════════════════════════════════════ */
  let dragSrcId = null;

  function renderTabs() {
    const bar    = document.getElementById('tabBar');
    const addBtn = bar.querySelector('.tab-add-btn');
    Array.from(bar.querySelectorAll('.tab')).forEach(el => el.remove());

    tabs.forEach(tab => {
      const el = document.createElement('div');
      el.className = 'tab' + (tab.id === activeId ? ' active' : '');
      el.dataset.id = tab.id;
      el.draggable = true;

      /* drag-to-reorder */
      el.addEventListener('dragstart', e => {
        dragSrcId = tab.id;
        e.dataTransfer.effectAllowed = 'move';
        setTimeout(() => el.style.opacity = '0.5', 0);
      });
      el.addEventListener('dragend', () => { el.style.opacity = ''; });
      el.addEventListener('dragover', e => {
        e.preventDefault();
        el.classList.add('drag-over');
      });
      el.addEventListener('dragleave', () => el.classList.remove('drag-over'));
      el.addEventListener('drop', e => {
        e.preventDefault();
        el.classList.remove('drag-over');
        if (!dragSrcId || dragSrcId === tab.id) return;
        flushActiveCode();
        const fromIdx = tabs.findIndex(t => t.id === dragSrcId);
        const toIdx   = tabs.findIndex(t => t.id === tab.id);
        const [moved] = tabs.splice(fromIdx, 1);
        tabs.splice(toIdx, 0, moved);
        saveTabs();
        renderTabs();
      });

      /* name span */
      const nameSpan = document.createElement('span');
      nameSpan.className = 'tab-name';
      nameSpan.textContent = tab.name;
      nameSpan.title = 'Double-click to rename';
      nameSpan.addEventListener('dblclick', e => {
        e.stopPropagation();
        startRename(tab.id, nameSpan);
      });

      /* close btn */
      const closeBtn = document.createElement('span');
      closeBtn.className = 'close-btn';
      closeBtn.textContent = '×';
      closeBtn.title = 'Close tab';
      closeBtn.addEventListener('click', e => {
        e.stopPropagation();
        closeTabById(tab.id);
      });

      el.appendChild(nameSpan);
      if (tabs.length > 1) el.appendChild(closeBtn);

      el.addEventListener('click', () => switchTab(tab.id));
      el.addEventListener('contextmenu', e => openCtxMenu(e, tab.id));

      bar.insertBefore(el, addBtn);
    });
  }

  function startRename(id, nameSpanOrNull) {
    const tab = tabs.find(t => t.id === id);
    if (!tab) return;

    /* If called from context menu we need to find the span */
    if (!nameSpanOrNull) {
      const el = document.querySelector(`.tab[data-id="${id}"] .tab-name`);
      if (el) nameSpanOrNull = el; else return;
    }

    const input = document.createElement('input');
    input.className = 'tab-name-input';
    input.value = tab.name;
    nameSpanOrNull.replaceWith(input);
    input.focus();
    input.select();

    function commit() {
      const val = input.value.trim() || tab.name;
      tab.name = val;
      saveTabs();
      renderTabs();
    }
    input.addEventListener('blur', commit);
    input.addEventListener('keydown', e => {
      if (e.key === 'Enter')  input.blur();
      if (e.key === 'Escape') { input.value = tab.name; input.blur(); }
    });
  }

  /* ═══════════════════════════════════════════════════════
     CONTEXT MENU
  ═══════════════════════════════════════════════════════ */
  let ctxTargetId = null;
  const ctxMenu = document.getElementById('ctxMenu');

  function openCtxMenu(e, id) {
    e.preventDefault();
    e.stopPropagation();
    ctxTargetId = id;
    ctxMenu.style.left = e.clientX + 'px';
    ctxMenu.style.top  = e.clientY + 'px';
    ctxMenu.classList.add('open');

    /* disable Close option if only 1 tab */
    ctxMenu.querySelectorAll('.ctx-item')[2].style.opacity = tabs.length <= 1 ? '0.4' : '1';
    ctxMenu.querySelectorAll('.ctx-item')[2].style.pointerEvents = tabs.length <= 1 ? 'none' : '';
  }

  function closeCtxMenu() { ctxMenu.classList.remove('open'); ctxTargetId = null; }
  document.addEventListener('click', closeCtxMenu);
  document.addEventListener('keydown', e => { if (e.key === 'Escape') closeCtxMenu(); });

  function ctxRename() {
    const id = ctxTargetId; closeCtxMenu();
    startRename(id, null);
  }
  function ctxDupe() {
    const id = ctxTargetId; closeCtxMenu();
    duplicateTab(id);
  }
  function ctxClose() {
    const id = ctxTargetId; closeCtxMenu();
    closeTabById(id);
  }

  /* ═══════════════════════════════════════════════════════
     TAB OPERATIONS
  ═══════════════════════════════════════════════════════ */
  function switchTab(id) {
    flushActiveCode();
    activeId = id;
    saveTabs();
    loadActiveCode();
    renderTabs();
    updateLineNumbers();
    run();
  }

  function addTab() {
    flushActiveCode();
    const id  = genId();
    const num = tabs.length + 1;
    tabs.push({ id, name: 'Tab ' + num, code: '<h1>Hello World</h1>' });
    activeId = id;
    saveTabs();
    loadActiveCode();
    renderTabs();
    updateLineNumbers();
    run();
  }

  function duplicateTab(id) {
    flushActiveCode();
    const src = tabs.find(t => t.id === id);
    if (!src) return;
    const newId = genId();
    const idx   = tabs.findIndex(t => t.id === id);
    tabs.splice(idx + 1, 0, { id: newId, name: src.name + ' (copy)', code: src.code });
    activeId = newId;
    saveTabs();
    loadActiveCode();
    renderTabs();
    updateLineNumbers();
    run();
  }

  function closeTabById(id) {
    if (tabs.length <= 1) return;
    const idx = tabs.findIndex(t => t.id === id);
    tabs.splice(idx, 1);
    if (activeId === id) {
      activeId = tabs[Math.min(idx, tabs.length - 1)].id;
      loadActiveCode();
      run();
    }
    saveTabs();
    renderTabs();
    updateLineNumbers();
  }

  function flushActiveCode() {
    const tab = activeTab();
    if (tab) tab.code = codeBox.value;
  }

  function loadActiveCode() {
    const tab = activeTab();
    if (tab) codeBox.value = tab.code;
  }

  /* ═══════════════════════════════════════════════════════
     TAB SEARCH
  ═══════════════════════════════════════════════════════ */
  function toggleTabSearch() {
    const dd = document.getElementById('tabSearchDropdown');
    const isOpen = dd.classList.toggle('open');
    if (isOpen) {
      filterTabSearch();
      document.getElementById('tabSearchInput').focus();
    }
  }

  function filterTabSearch() {
    const q    = document.getElementById('tabSearchInput').value.toLowerCase();
    const list = document.getElementById('tabSearchList');
    list.innerHTML = '';
    tabs.filter(t => t.name.toLowerCase().includes(q)).forEach(t => {
      const item = document.createElement('div');
      item.className = 'tab-search-item' + (t.id === activeId ? ' active-item' : '');
      item.textContent = t.name;
      item.addEventListener('click', () => {
        document.getElementById('tabSearchDropdown').classList.remove('open');
        switchTab(t.id);
        /* scroll tab into view */
        const el = document.querySelector(`.tab[data-id="${t.id}"]`);
        if (el) el.scrollIntoView({ inline: 'nearest' });
      });
      list.appendChild(item);
    });
    if (!list.children.length) {
      const empty = document.createElement('div');
      empty.className = 'tab-search-item';
      empty.style.color = '#475569';
      empty.textContent = 'No tabs found';
      list.appendChild(empty);
    }
  }

  /* Close tab-search when clicking outside */
  document.addEventListener('click', e => {
    const btn = document.getElementById('tabSearchBtn');
    const dd  = document.getElementById('tabSearchDropdown');
    if (!btn.contains(e.target) && !dd.contains(e.target)) {
      dd.classList.remove('open');
    }
  });

  /* ═══════════════════════════════════════════════════════
     SESSION EXPORT / IMPORT
  ═══════════════════════════════════════════════════════ */
  function exportSession() {
    flushActiveCode();
    const bundle = JSON.stringify({ tabs, activeId }, null, 2);
    const blob = new Blob([bundle], { type: 'application/json' });
    const a = document.createElement('a');
    a.href = URL.createObjectURL(blob);
    a.download = 'html-editor-session.json';
    a.click();
  }

  function importSession(event) {
    const file = event.target.files[0];
    if (!file) return;
    const reader = new FileReader();
    reader.onload = e => {
      try {
        const bundle = JSON.parse(e.target.result);
        if (!Array.isArray(bundle.tabs) || !bundle.tabs.length) throw new Error('bad');
        tabs     = bundle.tabs;
        activeId = bundle.activeId || tabs[0].id;
        saveTabs();
        loadActiveCode();
        renderTabs();
        updateLineNumbers();
        run();
      } catch {
        alert('Invalid session file.');
      }
    };
    reader.readAsText(file);
    event.target.value = '';
  }

  /* ═══════════════════════════════════════════════════════
     LINE NUMBERS
  ═══════════════════════════════════════════════════════ */
  function updateLineNumbers() {
    const lines = codeBox.value.split('\n').length;
    const ln    = document.getElementById('lineNumbers');
    let out = '';
    for (let i = 1; i <= lines; i++) out += i + '\n';
    ln.textContent = out;
    /* sync scroll */
    ln.scrollTop = codeBox.scrollTop;
  }

  /* ═══════════════════════════════════════════════════════
     DRAGGABLE DIVIDER
  ═══════════════════════════════════════════════════════ */
  (function() {
    const divider  = document.getElementById('divider');
    const mainArea = document.getElementById('mainArea');
    const leftPane = document.getElementById('leftPane');
    const rightPane= document.getElementById('rightPane');
    let dragging = false, startX = 0, startLeftW = 0;

    /* Restore saved split */
    const savedPct = parseFloat(localStorage.getItem('html_editor_split'));
    if (!isNaN(savedPct)) {
      leftPane.style.width  = savedPct + '%';
      rightPane.style.flex  = 'none';
      rightPane.style.width = (100 - savedPct) + '%';
    } else {
      leftPane.style.width  = '50%';
      rightPane.style.width = '50%';
    }

    divider.addEventListener('mousedown', e => {
      dragging   = true;
      startX     = e.clientX;
      startLeftW = leftPane.getBoundingClientRect().width;
      divider.classList.add('dragging');
      mainArea.classList.add('is-resizing');
      e.preventDefault();
    });

    document.addEventListener('mousemove', e => {
      if (!dragging) return;
      const totalW  = mainArea.getBoundingClientRect().width;
      const delta   = e.clientX - startX;
      let newLeftW  = startLeftW + delta;
      const minW    = 80;
      newLeftW      = Math.max(minW, Math.min(totalW - minW - 5, newLeftW));
      const pct     = (newLeftW / totalW) * 100;
      leftPane.style.width  = pct + '%';
      rightPane.style.width = (100 - pct - (5 / totalW * 100)) + '%';
      localStorage.setItem('html_editor_split', pct.toFixed(2));
    });

    document.addEventListener('mouseup', () => {
      if (!dragging) return;
      dragging = false;
      divider.classList.remove('dragging');
      mainArea.classList.remove('is-resizing');
    });
  })();

  /* ═══════════════════════════════════════════════════════
     ORIGINAL FEATURES (unchanged behaviour)
  ═══════════════════════════════════════════════════════ */
  let auto = true;
  const iframe  = document.getElementById('preview');
  const codeBox = document.getElementById('code');

  function save() {
    flushActiveCode();
    saveTabs();
    localStorage.setItem('html_code', codeBox.value);
  }

  function run() {
    save();
    const doc = iframe.contentDocument || iframe.contentWindow.document;
    doc.open();
    doc.write(codeBox.value);
    doc.close();
  }

  /* Tab key → insert spaces instead of shifting focus */
  codeBox.addEventListener('keydown', e => {
    if (e.key === 'Tab') {
      e.preventDefault();
      const s = codeBox.selectionStart;
      const v = codeBox.value;
      codeBox.value = v.slice(0, s) + '  ' + v.slice(codeBox.selectionEnd);
      codeBox.selectionStart = codeBox.selectionEnd = s + 2;
      if (auto) run();
    }
  });

  codeBox.addEventListener('input', () => {
    updateLineNumbers();
    if (auto) run();
  });

  codeBox.addEventListener('scroll', () => {
    document.getElementById('lineNumbers').scrollTop = codeBox.scrollTop;
  });

  function openTab() {
    const win = window.open('about:blank', '_blank');
    win.document.open();
    win.document.write(codeBox.value);
    win.document.close();
  }

  function resetCode() {
    codeBox.value = '';
    updateLineNumbers();
    run();
  }

  function toggleAuto() {
    auto = !auto;
    document.getElementById('autoToggle').classList.toggle('on', auto);
  }

  /* Export uses tab name as filename */
  function exportCode() {
    const name = (activeTab()?.name || 'index').replace(/[^a-z0-9_\-]/gi, '_');
    const blob = new Blob([codeBox.value], { type: 'text/html' });
    const a = document.createElement('a');
    a.href = URL.createObjectURL(blob);
    a.download = name + '.html';
    a.click();
  }

  function importCode(event) {
    const file = event.target.files[0];
    if (!file) return;
    const reader = new FileReader();
    reader.onload = e => {
      codeBox.value = e.target.result;
      updateLineNumbers();
      run();
    };
    reader.readAsText(file);
    event.target.value = '';
  }

  loadTabs();
